\section{Problème 1: Gradients et Hessiennes}

\subsection{Question 1}

Soit $f(\mathbf{x}) = \frac{1}{2}\mathbf{x}^\top \mathbf{A}\mathbf{x} + \mathbf{b}^\top\mathbf{x} + c$ avec $\mathbf{A}$ symétrique.

\textbf{Réponse~: (b)} $\nabla_{\mathbf{x}} f(\mathbf{x}) = \mathbf{A}\mathbf{x} + \mathbf{b}$.

\textbf{Justification.}
On applique les règles de dérivation matricielle terme à terme~:
\begin{itemize}
  \item Forme quadratique~: $\frac{\partial}{\partial \mathbf{x}}(\mathbf{x}^\top\mathbf{A}\mathbf{x}) = (\mathbf{A} + \mathbf{A}^\top)\mathbf{x} = 2\mathbf{A}\mathbf{x}$ car $\mathbf{A}^\top = \mathbf{A}$.
  \item Terme linéaire~: $\frac{\partial}{\partial \mathbf{x}}(\mathbf{b}^\top\mathbf{x}) = \mathbf{b}$ (dérivée de $\mathbf{b}^\top\mathbf{A}\mathbf{x}$ avec $\mathbf{A} = \mathbf{I}$).
  \item Constante~: $\frac{\partial c}{\partial \mathbf{x}} = \mathbf{0}$.
\end{itemize}
D'où
\[
\nabla_{\mathbf{x}} f(\mathbf{x}) = \tfrac{1}{2}\cdot 2\mathbf{A}\mathbf{x} + \mathbf{b} = \boxed{\mathbf{A}\mathbf{x} + \mathbf{b}}.
\]

\subsection{Question 2}

Même fonction $f(\mathbf{x}) = \frac{1}{2}\mathbf{x}^\top \mathbf{A}\mathbf{x} + \mathbf{b}^\top\mathbf{x} + c$.

\textbf{Réponse~: (a)} $\nabla^2_{\mathbf{x}} f(\mathbf{x}) = \mathbf{A}$.

\textbf{Justification.}
Le gradient est $\nabla f(\mathbf{x}) = \mathbf{A}\mathbf{x} + \mathbf{b}$. On dérive une seconde fois~:
\[
\nabla^2_{\mathbf{x}} f(\mathbf{x}) = \frac{\partial}{\partial \mathbf{x}}(\mathbf{A}\mathbf{x} + \mathbf{b}) = \boxed{\mathbf{A}}.
\]

\subsection{Question 3}

Soit $f(\mathbf{x}) = e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}$ avec $\mathbf{A}$ symétrique.

\textbf{Réponse~: (b)} $\nabla f(\mathbf{x}) = 2e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\mathbf{A}\mathbf{x}$.

\textbf{Justification.}
On pose $h(\mathbf{x}) = \mathbf{x}^\top\mathbf{A}\mathbf{x}$ de sorte que $f = e^{h}$. Par la règle de la dérivée en chaîne~:
\[
\nabla f = e^{h}\,\nabla h.
\]
Or $\nabla h = (\mathbf{A}+\mathbf{A}^\top)\mathbf{x} = 2\mathbf{A}\mathbf{x}$ ($\mathbf{A}$ symétrique), d'où
\[
\nabla f(\mathbf{x}) = \boxed{2e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\,\mathbf{A}\mathbf{x}}.
\]

\subsection{Question 4}

Même fonction $f(\mathbf{x}) = e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}$.

\textbf{Réponse~: (a)} $\nabla^2_{\mathbf{x}} f(\mathbf{x}) = e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}(2\mathbf{A} + 4\mathbf{A}\mathbf{x}\mathbf{x}^\top\mathbf{A})$.

\textbf{Justification.}
On a $\nabla f(\mathbf{x}) = 2e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\,\mathbf{A}\mathbf{x}$. La Hessienne est le Jacobien du gradient. Par la règle du produit matriciel appliquée à $\mathbf{g}(\mathbf{x}) = u(\mathbf{x})\,\mathbf{v}(\mathbf{x})$ avec $u = 2e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}$ (scalaire) et $\mathbf{v} = \mathbf{A}\mathbf{x}$ (vecteur)~:
\[
\frac{\partial \mathbf{g}}{\partial \mathbf{x}} = \mathbf{v}\,\frac{\partial u}{\partial \mathbf{x}}^\top + u\,\frac{\partial \mathbf{v}}{\partial \mathbf{x}}.
\]
On a $\frac{\partial u}{\partial \mathbf{x}} = 2u\,\mathbf{A}\mathbf{x}$ (d'après Q3 sans le facteur 2 initial, soit $4e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\mathbf{A}\mathbf{x}$) et $\frac{\partial \mathbf{v}}{\partial \mathbf{x}} = \mathbf{A}$. Donc~:
\[
\nabla^2 f = (\mathbf{A}\mathbf{x})(4e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\mathbf{A}\mathbf{x})^\top + 2e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\mathbf{A} % chktex 3
= e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}\bigl(4\mathbf{A}\mathbf{x}\mathbf{x}^\top\mathbf{A} + 2\mathbf{A}\bigr).
\]
\[
\boxed{\nabla^2_{\mathbf{x}} f(\mathbf{x}) = e^{\mathbf{x}^\top\mathbf{A}\mathbf{x}}(2\mathbf{A} + 4\mathbf{A}\mathbf{x}\mathbf{x}^\top\mathbf{A})}.
\]

\subsection{Question 5}

Soit $f(\mathbf{x}) = \|\mathbf{A}\mathbf{x}\|^2 = (\mathbf{A}\mathbf{x})^\top(\mathbf{A}\mathbf{x}) = \mathbf{x}^\top\mathbf{A}^\top\mathbf{A}\mathbf{x}$. % chktex 3

\textbf{Réponse~: (a)} $\nabla_{\mathbf{x}} f(\mathbf{x}) = 2\mathbf{A}^\top\mathbf{A}\mathbf{x}$.

\textbf{Justification.}
Posons $\mathbf{B} = \mathbf{A}^\top\mathbf{A}$. La matrice $\mathbf{B}$ est symétrique car $\mathbf{B}^\top = (\mathbf{A}^\top\mathbf{A})^\top = \mathbf{A}^\top\mathbf{A} = \mathbf{B}$. % chktex 3
Alors $f(\mathbf{x}) = \mathbf{x}^\top\mathbf{B}\mathbf{x}$ et
\[
\nabla f = (\mathbf{B}+\mathbf{B}^\top)\mathbf{x} = 2\mathbf{B}\mathbf{x} = \boxed{2\mathbf{A}^\top\mathbf{A}\mathbf{x}}.
\]

\subsection{Question 6}

Même fonction $f(\mathbf{x}) = \|\mathbf{A}\mathbf{x}\|^2 = \mathbf{x}^\top\mathbf{A}^\top\mathbf{A}\mathbf{x}$.

\textbf{Réponse~: (a)} $\nabla^2_{\mathbf{x}} f(\mathbf{x}) = 2\mathbf{A}^\top\mathbf{A}$.

\textbf{Justification.}
Le gradient est $\nabla f(\mathbf{x}) = 2\mathbf{A}^\top\mathbf{A}\mathbf{x}$. On dérive une seconde fois~:
\[
\nabla^2_{\mathbf{x}} f(\mathbf{x}) = \frac{\partial}{\partial \mathbf{x}}(2\mathbf{A}^\top\mathbf{A}\mathbf{x}) = \boxed{2\mathbf{A}^\top\mathbf{A}}.
\]

\section{Problème 2: Optimisation, Descentes de Gradient vs Newton et analyse spectrale}

On considère la fonction définie sur $x_1 > 0$, $x_2 > 0$~:
\[
f(\mathbf{x}) = \tfrac{1}{2}\mathbf{x}^\top\mathbf{A}\mathbf{x} - \mathbf{b}^\top\mathbf{x} - \mu\bigl(\log(x_1) + \log(x_2)\bigr),
\]
avec
$\mathbf{A} = \begin{pmatrix} 4 & 1 \\ 1 & 3 \end{pmatrix}$,
$\mathbf{b} = \begin{pmatrix} 1 \\ 2 \end{pmatrix}$,
$\mu = 0{,}1$,
$\mathbf{x}_0 = \begin{pmatrix} 1 \\ 1 \end{pmatrix}$.

\subsection{a) Gradient} % chktex 9 chktex 10

\textbf{Réponse~: (ii)} $\displaystyle \nabla f(\mathbf{x}) = \mathbf{A}\mathbf{x} - \mathbf{b} - \mu\begin{pmatrix} 1/x_1 \\ 1/x_2 \end{pmatrix}$.

\textbf{Justification.}
On dérive chaque terme~:
\begin{itemize}
  \item $\frac{\partial}{\partial \mathbf{x}}\bigl(\tfrac{1}{2}\mathbf{x}^\top\mathbf{A}\mathbf{x}\bigr) = \mathbf{A}\mathbf{x}$ ($\mathbf{A}$ symétrique).
  \item $\frac{\partial}{\partial \mathbf{x}}\bigl(-\mathbf{b}^\top\mathbf{x}\bigr) = -\mathbf{b}$.
  \item $\frac{\partial}{\partial x_i}\bigl(-\mu\log x_i\bigr) = -\mu/x_i$.
\end{itemize}
D'où $\nabla f(\mathbf{x}) = \boxed{\mathbf{A}\mathbf{x} - \mathbf{b} - \mu\begin{pmatrix} 1/x_1 \\ 1/x_2 \end{pmatrix}}$.

\subsection{b) Hessienne} % chktex 9 chktex 10

\textbf{Réponse~: (ii)} $\displaystyle H(\mathbf{x}) = \mathbf{A} + \mu\,\text{diag}\!\left(\frac{1}{x_1^2},\, \frac{1}{x_2^2}\right)$.

\textbf{Justification.}
On dérive le gradient composante par composante~:
\begin{itemize}
  \item $\frac{\partial}{\partial \mathbf{x}}(\mathbf{A}\mathbf{x} - \mathbf{b}) = \mathbf{A}$.
  \item $\frac{\partial}{\partial x_i}\bigl(-\mu/x_i\bigr) = \mu/x_i^2$ sur la diagonale, $0$ ailleurs.
\end{itemize}
D'où $H(\mathbf{x}) = \boxed{\mathbf{A} + \mu\,\text{diag}\!\left(\frac{1}{x_1^2},\, \frac{1}{x_2^2}\right)}$.

\subsection{c) Comportement près de la frontière} % chktex 9 chktex 10

\textbf{Réponse~: (iv)} Les trois affirmations sont vraies.

\textbf{Justification.}
Quand $x_1 \to 0^+$~:
\begin{enumerate}
  \item[(i)] $-\mu\log(x_1) \to +\infty$, donc $f(\mathbf{x}) \to +\infty$. \textbf{Vrai}: la barrière logarithmique empêche $\mathbf{x}$ de sortir du domaine.
  \item[(ii)] Le gradient contient $-\mu/x_1 \to -\infty$ en magnitude. \textbf{Vrai}: le gradient repousse $\mathbf{x}$ loin de la frontière.
  \item[(iii)] La Hessienne contient $\mu/x_1^2 \to +\infty$, augmentant la courbure locale. \textbf{Vrai}: la fonction devient très «~raide~» près de la frontière.
\end{enumerate}

\subsection{d) Modèle quadratique local} % chktex 9 chktex 10

\textbf{Réponse~: (ii)} $H\mathbf{p} = -\mathbf{g}$.

\textbf{Justification.}
Le modèle quadratique est~:
\[
f(\mathbf{x}_k + \mathbf{p}) \approx f(\mathbf{x}_k) + \mathbf{g}^\top\mathbf{p} + \tfrac{1}{2}\mathbf{p}^\top H\mathbf{p}.
\]
On minimise par rapport à $\mathbf{p}$ en annulant le gradient~:
\[
\frac{\partial}{\partial \mathbf{p}}\bigl(\mathbf{g}^\top\mathbf{p} + \tfrac{1}{2}\mathbf{p}^\top H\mathbf{p}\bigr) = \mathbf{g} + H\mathbf{p} = \mathbf{0}
\quad\Longrightarrow\quad
\boxed{H\mathbf{p} = -\mathbf{g}}.
\]
C'est l'équation de Newton.

\subsection{g) Analyse spectrale} % chktex 9 chktex 10

\textbf{(i) Valeurs propres et vecteurs propres.}

Pour $\mathbf{x}^{(1)} = (1, 1)^\top$~: % chktex 3
\[
H(\mathbf{x}^{(1)}) = \begin{pmatrix} 4{,}1 & 1 \\ 1 & 3{,}1 \end{pmatrix}, \quad
\lambda_{\min} = 2{,}482,\;\; \lambda_{\max} = 4{,}718.
\]

Pour $\mathbf{x}^{(2)} = (0{,}2,\; 0{,}2)^\top$~: % chktex 3
\[
H(\mathbf{x}^{(2)}) = \begin{pmatrix} 6{,}5 & 1 \\ 1 & 5{,}5 \end{pmatrix}, \quad
\lambda_{\min} = 4{,}882,\;\; \lambda_{\max} = 7{,}118.
\]

Les vecteurs propres associés sont identiques dans les deux cas~:
$\mathbf{v}_{\min} \approx (0{,}526,\; {-0{,}851})^\top$, % chktex 3
$\mathbf{v}_{\max} \approx (-0{,}851,\; {-0{,}526})^\top$. % chktex 3

\textbf{(ii) Signification de $\kappa(\mathbf{x})$.}
Le nombre de conditionnement $\kappa = \lambda_{\max}/\lambda_{\min}$ mesure l'\textbf{anisotropie} de la Hessienne.
Un $\kappa$ élevé signifie que les courbes de niveau sont très allongées (ellipses aplaties), ce qui ralentit la descente de gradient qui zigzague entre les parois de la «~vallée~».
Un $\kappa$ proche de $1$ signifie des courbes de niveau quasi circulaires, favorable à GD\@.

\textbf{(iii) Conditionnement aux deux points.}
\[
\kappa(\mathbf{x}^{(1)}) = \frac{4{,}718}{2{,}482} \approx 1{,}901, \qquad
\kappa(\mathbf{x}^{(2)}) = \frac{7{,}118}{4{,}882} \approx 1{,}458.
\]

Les deux valeurs sont proches de $1$, ce qui indique un problème \textbf{bien conditionné} aux deux points.
Le conditionnement est même meilleur en $\mathbf{x}^{(2)}$ ($\kappa \approx 1{,}46$) qu'en $\mathbf{x}^{(1)}$ ($\kappa \approx 1{,}90$), malgré la proximité de la frontière.
Cela explique la convergence raisonnablement rapide de GD (67 itérations) pour ce problème.
Près de la frontière, la barrière logarithmique ajoute de la courbure (termes $\mu/x_i^2$), ce qui augmente les deux valeurs propres et peut améliorer le conditionnement.

\section{Problème 3: Estimation par la Méthode du Maximum de Vraisemblance et Probabilités des Estimateurs}

Soit $X_1, X_2, \ldots, X_n$ un échantillon aléatoire issu d'une \textbf{loi de Poisson} de paramètre $\lambda$:
\[
P(X = k) = \frac{\lambda^k e^{-\lambda}}{k!}, \quad k = 0, 1, 2, \ldots
\]

\subsection{a) Estimateur du maximum de vraisemblance de $\lambda$} % chktex 9 chktex 10

\textbf{Fonction de vraisemblance.}
\[
L(\lambda; \mathbf{X}) = \prod_{i=1}^{n} \frac{\lambda^{X_i} e^{-\lambda}}{X_i!}
= \frac{\lambda^{\sum_{i=1}^n X_i} \, e^{-n\lambda}}{\prod_{i=1}^n X_i!}.
\]

\textbf{Log-vraisemblance.}
\[
\ell(\lambda) = \log L(\lambda) = \left(\sum_{i=1}^n X_i\right) \log\lambda - n\lambda - \sum_{i=1}^n \log(X_i!).
\]

\textbf{Maximisation.} On dérive par rapport à $\lambda$ et on annule:
\[
\frac{d\ell}{d\lambda} = \frac{\sum_{i=1}^n X_i}{\lambda} - n = 0
\quad \Longrightarrow \quad
\boxed{\hat{\lambda}_{\text{EMV}} = \bar{X} = \frac{1}{n}\sum_{i=1}^n X_i.}
\]

\textbf{Vérification (condition du second ordre).}
$\displaystyle \frac{d^2\ell}{d\lambda^2} = -\frac{\sum_{i=1}^n X_i}{\lambda^2} < 0$,
donc $\hat{\lambda}$ est bien un maximum.

\subsection{b) Biais de l'estimateur} % chktex 9 chktex 10

On calcule l'espérance de $\hat{\lambda}$:
\[
\mathbb{E}[\hat{\lambda}]
= \mathbb{E}\!\left[\frac{1}{n}\sum_{i=1}^n X_i\right]
= \frac{1}{n}\sum_{i=1}^n \mathbb{E}[X_i]
= \frac{1}{n} \cdot n\lambda
= \lambda.
\]
Le biais est $b(\hat{\lambda}) = \mathbb{E}[\hat{\lambda}] - \lambda = 0$.
L'estimateur $\hat{\lambda} = \bar{X}$ est donc \textbf{non biaisé}.

\subsection{c) Consistance de l'estimateur} % chktex 9 chktex 10

Pour une variable de Poisson, $\text{Var}(X_i) = \lambda$. On a donc:
\[
\text{Var}(\hat{\lambda})
= \text{Var}(\bar{X})
= \frac{1}{n^2}\sum_{i=1}^n \text{Var}(X_i)
= \frac{n\lambda}{n^2}
= \frac{\lambda}{n}.
\]

L'erreur quadratique moyenne est:
\[
EQM(\hat{\lambda}_n) = \text{Var}(\hat{\lambda}_n) + [b(\hat{\lambda}_n)]^2 = \frac{\lambda}{n} + 0 = \frac{\lambda}{n}. % chktex 3
\]

Comme $\displaystyle \lim_{n \to \infty} EQM(\hat{\lambda}_n) = \lim_{n \to \infty} \frac{\lambda}{n} = 0$,
l'estimateur $\hat{\lambda}$ est \textbf{consistant}.

\subsection{d) Application numérique} % chktex 9 chktex 10

On dispose des fréquences observées suivantes:

\begin{center}
\begin{tabular}{c|cccccc} % chktex 44
$x$ & 0 & 1 & 2 & 3 & 4 & 5 \\ \hline % chktex 44
$f_X(x)$ & 7 & 14 & 12 & 13 & 6 & 3
\end{tabular}
\end{center}

\textbf{(i) EMV de $\lambda$.}
Le nombre total d'observations est $n = 7+14+12+13+6+3 = 55$ et
\[
\sum_{i=1}^n X_i = 0(7) + 1(14) + 2(12) + 3(13) + 4(6) + 5(3) = 116.
\]
D'où
\[
\hat{\lambda} = \bar{X} = \frac{116}{55} \approx 2{,}1091.
\]

\textbf{(ii) Estimation de $P(X = 2)$.}
On substitue $\hat{\lambda}$ dans la loi de Poisson:
\[
\hat{P}(X = 2) = \frac{\hat{\lambda}^2\, e^{-\hat{\lambda}}}{2!}
= \frac{(116/55)^2\, e^{-116/55}}{2} % chktex 3
\approx 0{,}2699.
\]

\section{Problème 5: Loi des Grands Nombres et Théorème Central Limite}

On simule une variable aléatoire $X \sim \text{Bernoulli}(p = 0{,}5)$ et on étudie le comportement de la moyenne empirique $\overline{X}_n = \frac{1}{n}\sum_{i=1}^{n} X_i$ lorsque $n$ augmente.

\subsection{Question 1}

\textbf{Influence de $n$ sur la dispersion de $\overline{X}_n$ (loi faible des grands nombres).}

On répète $2000$ fois le calcul de $\overline{X}_n$ pour $n \in \{50,\; 500,\; 5000\}$ et on trace l'histogramme des moyennes obtenues.

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\textwidth]{../output/5_1.png}
  \caption{Distribution de $\overline{X}_n$ pour $n = 50$, $500$ et $5000$ ($2000$ répétitions) (output/5\_1.png).}%
  \label{fig:lfgn}
\end{figure}

La figure~\ref{fig:lfgn} montre que lorsque $n$ augmente, la distribution de $\overline{X}_n$ se concentre de plus en plus autour de $\mathbb{E}[X] = 0{,}5$.
Concrètement, la variance de $\overline{X}_n$ est $\text{Var}(\overline{X}_n) = \frac{p(1-p)}{n} = \frac{0{,}25}{n}$, qui décroît en $1/n$.
Pour $n = 50$ l'histogramme est large (écart-type $\approx 0{,}071$), pour $n = 500$ il est nettement plus étroit ($\approx 0{,}022$), et pour $n = 5000$ les valeurs sont quasiment toutes égales à $0{,}5$ ($\approx 0{,}007$).
Cela illustre la \textbf{loi faible des grands nombres}: $\overline{X}_n \xrightarrow{P} \mathbb{E}[X]$ lorsque $n \to \infty$.

\subsection{Question 2}

\textbf{Trajectoires individuelles de $\overline{X}_n$ (loi forte des grands nombres).}

On fixe $n = 1000$ et on trace les trajectoires de la moyenne cumulative $\overline{X}_k$ ($k = 1, \dots, n$) pour $N_{\text{rep}} \in \{1,\; 5,\; 10,\; 1000\}$ répétitions.

La figure~\ref{fig:lfortegn} montre que, quel que soit le nombre de répétitions, \textbf{chaque trajectoire individuelle converge vers} $\mathbb{E}[X] = 0{,}5$ lorsque $n$ grandit.
Même avec une seule trajectoire ($N_{\text{rep}} = 1$), la moyenne cumulative se stabilise autour de $0{,}5$ pour $n$ suffisamment grand.
En augmentant le nombre de répétitions, on observe un faisceau de trajectoires qui se resserre autour de $0{,}5$: les fluctuations initiales sont grandes, mais elles s'amortissent toutes de la même façon.
Cela illustre la \textbf{loi forte des grands nombres}: $\overline{X}_n \xrightarrow{\text{p.s.}} \mathbb{E}[X]$, c'est-à-dire que la convergence a lieu pour \emph{presque toute} réalisation, et non seulement en probabilité.

\subsection{Question 3}

\textbf{Convergence vers la densité normale (théorème central limite).}

On calcule $\overline{X}_n$ sur $500$ répétitions pour $n \in \{100,\; 1000,\; 10\,000\}$ et on superpose la densité de la loi $\mathcal{N}\!\left(0{,}5,\; \frac{p(1-p)}{n}\right)$.

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\textwidth]{../output/5_3.png}
  \caption{Distribution de $\overline{X}_n$ comparée à la densité normale théorique pour $n = 100$, $1000$ et $10\,000$ (output/5\_3.png).}%
  \label{fig:tcl}
\end{figure}

La figure~\ref{fig:tcl} montre que l'histogramme des moyennes empiriques épouse de mieux en mieux la courbe de la densité normale (en rouge) à mesure que $n$ augmente.
Dès $n = 100$ la forme est approximativement gaussienne; pour $n = 1000$ et surtout $n = 10\,000$ la correspondance est excellente.
L'histogramme converge donc vers la densité de la loi \textbf{normale} $\mathcal{N}\!\left(\mu,\; \sigma^2/n\right)$ avec $\mu = p = 0{,}5$ et $\sigma^2 = p(1-p) = 0{,}25$.
Cela illustre le \textbf{théorème central limite}:
\[
\sqrt{n}\,\frac{\overline{X}_n - \mu}{\sigma} \xrightarrow{\mathcal{L}} \mathcal{N}(0, 1).
\]
