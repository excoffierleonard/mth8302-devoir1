\section{Problème 3: Estimation par la Méthode du Maximum de Vraisemblance et Probabilités des Estimateurs}

Soit $X_1, X_2, \ldots, X_n$ un échantillon aléatoire issu d'une \textbf{loi de Poisson} de paramètre $\lambda$:
\[
P(X = k) = \frac{\lambda^k e^{-\lambda}}{k!}, \quad k = 0, 1, 2, \ldots
\]

\subsection{a) Estimateur du maximum de vraisemblance de $\lambda$} % chktex 9 chktex 10

\textbf{Fonction de vraisemblance.}
\[
L(\lambda; \mathbf{X}) = \prod_{i=1}^{n} \frac{\lambda^{X_i} e^{-\lambda}}{X_i!}
= \frac{\lambda^{\sum_{i=1}^n X_i} \, e^{-n\lambda}}{\prod_{i=1}^n X_i!}.
\]

\textbf{Log-vraisemblance.}
\[
\ell(\lambda) = \log L(\lambda) = \left(\sum_{i=1}^n X_i\right) \log\lambda - n\lambda - \sum_{i=1}^n \log(X_i!).
\]

\textbf{Maximisation.} On dérive par rapport à $\lambda$ et on annule:
\[
\frac{d\ell}{d\lambda} = \frac{\sum_{i=1}^n X_i}{\lambda} - n = 0
\quad \Longrightarrow \quad
\boxed{\hat{\lambda}_{\text{EMV}} = \bar{X} = \frac{1}{n}\sum_{i=1}^n X_i.}
\]

\textbf{Vérification (condition du second ordre).}
$\displaystyle \frac{d^2\ell}{d\lambda^2} = -\frac{\sum_{i=1}^n X_i}{\lambda^2} < 0$,
donc $\hat{\lambda}$ est bien un maximum.

\subsection{b) Biais de l'estimateur} % chktex 9 chktex 10

On calcule l'espérance de $\hat{\lambda}$:
\[
\mathbb{E}[\hat{\lambda}]
= \mathbb{E}\!\left[\frac{1}{n}\sum_{i=1}^n X_i\right]
= \frac{1}{n}\sum_{i=1}^n \mathbb{E}[X_i]
= \frac{1}{n} \cdot n\lambda
= \lambda.
\]
Le biais est $b(\hat{\lambda}) = \mathbb{E}[\hat{\lambda}] - \lambda = 0$.
L'estimateur $\hat{\lambda} = \bar{X}$ est donc \textbf{non biaisé}.

\subsection{c) Consistance de l'estimateur} % chktex 9 chktex 10

Pour une variable de Poisson, $\text{Var}(X_i) = \lambda$. On a donc:
\[
\text{Var}(\hat{\lambda})
= \text{Var}(\bar{X})
= \frac{1}{n^2}\sum_{i=1}^n \text{Var}(X_i)
= \frac{n\lambda}{n^2}
= \frac{\lambda}{n}.
\]

L'erreur quadratique moyenne est:
\[
EQM(\hat{\lambda}_n) = \text{Var}(\hat{\lambda}_n) + [b(\hat{\lambda}_n)]^2 = \frac{\lambda}{n} + 0 = \frac{\lambda}{n}. % chktex 3
\]

Comme $\displaystyle \lim_{n \to \infty} EQM(\hat{\lambda}_n) = \lim_{n \to \infty} \frac{\lambda}{n} = 0$,
l'estimateur $\hat{\lambda}$ est \textbf{consistant}.

\subsection{d) Application numérique} % chktex 9 chktex 10

On dispose des fréquences observées suivantes:

\begin{center}
\begin{tabular}{c|cccccc} % chktex 44
$x$ & 0 & 1 & 2 & 3 & 4 & 5 \\ \hline % chktex 44
$f_X(x)$ & 7 & 14 & 12 & 13 & 6 & 3
\end{tabular}
\end{center}

\textbf{(i) EMV de $\lambda$.}
Le nombre total d'observations est $n = 7+14+12+13+6+3 = 55$ et
\[
\sum_{i=1}^n X_i = 0(7) + 1(14) + 2(12) + 3(13) + 4(6) + 5(3) = 116.
\]
D'où
\[
\hat{\lambda} = \bar{X} = \frac{116}{55} \approx 2{,}1091.
\]

\textbf{(ii) Estimation de $P(X = 2)$.}
On substitue $\hat{\lambda}$ dans la loi de Poisson:
\[
\hat{P}(X = 2) = \frac{\hat{\lambda}^2\, e^{-\hat{\lambda}}}{2!}
= \frac{(116/55)^2\, e^{-116/55}}{2} % chktex 3
\approx 0{,}2699.
\]

\section{Problème 4: Test F de Fisher et Intervalle de Confiance}

\subsection{a) Test F de Fisher} % chktex 9 chktex 10
\textbf{Hypothèses.} On teste
\(H_0: \sigma_A^2 = \sigma_B^2\) contre \(H_1: \sigma_A^2 \neq \sigma_B^2\) (test bilatéral), avec \(\alpha = 0.05\).

Les variances d'échantillon calculées sont \(s_A^2 = 16.988889\) et \(s_B^2 = 10.266667\), d'où la statistique \(F_{\text{stat}} = s_A^2 / s_B^2 = 1.654762\).
Les valeurs critiques pour \((\nu_1, \nu_2) = (9, 9)\) sont \(F_{0.025} = 0.248386\) et \(F_{0.975} = 4.025994\), et la p-valeur vaut \(0.464710\).
Comme \(F_{\text{stat}}\) est dans la région d'acceptation et \(p > 0.05\), on ne rejette pas \(H_0\): il n'y a pas d'évidence d'une différence de variances au seuil de 5\%.

  \caption{Distribution \(F(9,9)\) avec les zones critiques et la statistique observée \(F_{\text{stat}}\) (output/4.png).}%
  \label{fig:fisher-f}

La figure~\ref{fig:fisher-f} montre la densité \(F(9,9)\), les régions critiques (en rouge) et la position de \(F_{\text{stat}}\), qui se situe hors des zones de rejet.
\subsection{c) Intervalle de confiance} % chktex 9 chktex 10
\textbf{Intervalle de confiance à 95\% pour \(\sigma_A^2 / \sigma_B^2\).}
En utilisant les quantiles \(F_{0.025}\) et \(F_{0.975}\), l'intervalle est
\[
\left[\frac{F_{\text{stat}}}{F_{0.975}}, \frac{F_{\text{stat}}}{F_{0.025}}\right]=[0.411019,\; 6.662062].
\]

\section{Problème 5: Loi des Grands Nombres et Théorème Central Limite}

On simule une variable aléatoire $X \sim \text{Bernoulli}(p = 0{,}5)$ et on étudie le comportement de la moyenne empirique $\overline{X}_n = \frac{1}{n}\sum_{i=1}^{n} X_i$ lorsque $n$ augmente.

\subsection{Question 1}

\textbf{Influence de $n$ sur la dispersion de $\overline{X}_n$ (loi faible des grands nombres).}

On répète $2000$ fois le calcul de $\overline{X}_n$ pour $n \in \{50,\; 500,\; 5000\}$ et on trace l'histogramme des moyennes obtenues.

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\textwidth]{../output/5_1.png}
  \caption{Distribution de $\overline{X}_n$ pour $n = 50$, $500$ et $5000$ ($2000$ répétitions) (output/5\_1.png).}%
  \label{fig:lfgn}
\end{figure}

La figure~\ref{fig:lfgn} montre que lorsque $n$ augmente, la distribution de $\overline{X}_n$ se concentre de plus en plus autour de $\mathbb{E}[X] = 0{,}5$.
Concrètement, la variance de $\overline{X}_n$ est $\text{Var}(\overline{X}_n) = \frac{p(1-p)}{n} = \frac{0{,}25}{n}$, qui décroît en $1/n$.
Pour $n = 50$ l'histogramme est large (écart-type $\approx 0{,}071$), pour $n = 500$ il est nettement plus étroit ($\approx 0{,}022$), et pour $n = 5000$ les valeurs sont quasiment toutes égales à $0{,}5$ ($\approx 0{,}007$).
Cela illustre la \textbf{loi faible des grands nombres}: $\overline{X}_n \xrightarrow{P} \mathbb{E}[X]$ lorsque $n \to \infty$.

\subsection{Question 2}

\textbf{Trajectoires individuelles de $\overline{X}_n$ (loi forte des grands nombres).}

On fixe $n = 1000$ et on trace les trajectoires de la moyenne cumulative $\overline{X}_k$ ($k = 1, \dots, n$) pour $N_{\text{rep}} \in \{1,\; 5,\; 10,\; 1000\}$ répétitions.

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\textwidth]{../output/5_2.png}
  \caption{Trajectoires de $\overline{X}_n$ en fonction de $n$ pour différents nombres de répétitions (output/5\_2.png).}%
  \label{fig:lfortegn}
\end{figure}

La figure~\ref{fig:lfortegn} montre que, quel que soit le nombre de répétitions, \textbf{chaque trajectoire individuelle converge vers} $\mathbb{E}[X] = 0{,}5$ lorsque $n$ grandit.
Même avec une seule trajectoire ($N_{\text{rep}} = 1$), la moyenne cumulative se stabilise autour de $0{,}5$ pour $n$ suffisamment grand.
En augmentant le nombre de répétitions, on observe un faisceau de trajectoires qui se resserre autour de $0{,}5$: les fluctuations initiales sont grandes, mais elles s'amortissent toutes de la même façon.
Cela illustre la \textbf{loi forte des grands nombres}: $\overline{X}_n \xrightarrow{\text{p.s.}} \mathbb{E}[X]$, c'est-à-dire que la convergence a lieu pour \emph{presque toute} réalisation, et non seulement en probabilité.

\subsection{Question 3}

\textbf{Convergence vers la densité normale (théorème central limite).}

On calcule $\overline{X}_n$ sur $500$ répétitions pour $n \in \{100,\; 1000,\; 10\,000\}$ et on superpose la densité de la loi $\mathcal{N}\!\left(0{,}5,\; \frac{p(1-p)}{n}\right)$.

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\textwidth]{../output/5_3.png}
  \caption{Distribution de $\overline{X}_n$ comparée à la densité normale théorique pour $n = 100$, $1000$ et $10\,000$ (output/5\_3.png).}%
  \label{fig:tcl}
\end{figure}

La figure~\ref{fig:tcl} montre que l'histogramme des moyennes empiriques épouse de mieux en mieux la courbe de la densité normale (en rouge) à mesure que $n$ augmente.
Dès $n = 100$ la forme est approximativement gaussienne; pour $n = 1000$ et surtout $n = 10\,000$ la correspondance est excellente.
L'histogramme converge donc vers la densité de la loi \textbf{normale} $\mathcal{N}\!\left(\mu,\; \sigma^2/n\right)$ avec $\mu = p = 0{,}5$ et $\sigma^2 = p(1-p) = 0{,}25$.
Cela illustre le \textbf{théorème central limite}:
\[
\sqrt{n}\,\frac{\overline{X}_n - \mu}{\sigma} \xrightarrow{\mathcal{L}} \mathcal{N}(0, 1).
\]
